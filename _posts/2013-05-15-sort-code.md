#include <stdio.h>
#include  <time.h>

static int sort_arr[]  = {34,52,69,81,11,47,96,58,77,44,33,99,28,22,14,11};

void swap(int *a, int *b){
    int tmp = *a;
    *a = *b;
    *b = tmp;
}


//O(n^2）
void bubble_sort(int *arr, int length){
    for (int i=0; i<length-1; i++) {
        for (int j=0; j<length-1-i; j++) {
            if (arr[j]>arr[j+1]) {
                swap(&arr[j], &arr[j+1]);
            }
        }
    }
}

//O(n^2)
void select_sort(int *arr, int length){
    for (int i=0; i<length-1; i++) {
        int min = i;
        for (int j=i+1; j<length; j++) {
            if (arr[j]<arr[min]) {
                min = j;
            }
        }
        if (min!=i) {
            swap(&arr[i], &arr[min]);
        }
    }
}

//O(n^2)
void insert_sort(int *arr, int length){
    for (int i=1; i<length; i++) {
        for (int j=i; j>0; j--) {
            if (arr[j-1]>arr[j]) {
                swap(&arr[j-1], &arr[j]);
            }
        }
    }
}


void quick_sort(int *arr, int left, int right){
    int i = left,
    j = right,
    temp = arr[left];

    while (i<j) {
        while (i<j && temp<=arr[j]) {
            j--;
        }
        arr[i] = arr[j];

        while (i<j && temp>=arr[i]) {
            i++;
        }
        arr[j] = arr[i];
    }

    arr[i] = temp;

    if (i>left) {
        quick_sort(arr, left, i-1);
    }
    if (j<right) {
        quick_sort(arr, i+1, right);
    }
}
